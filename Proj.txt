  <script>
    let intervalId = null;
    let phase = 0; // 0 = abrir, 1 = fechar
    let remaining = 3;

    function renderStatus() {
      const phaseLabel = phase === 0 ? "ABRIR a mão" : "FECHAR a mão";
      document.getElementById("status").textContent =
        "Fase: " + phaseLabel + " | Restam: " + remaining + "s";
    }

    function tick() {
      if (remaining <= 1) {
        // troca fase
        phase = (phase + 1) % 2;
        remaining = 3;
      } else {
        remaining -= 1;
      }
      renderStatus();
    }

    function startTimer() {
      stopTimer();
      phase = 0;
      remaining = 3;
      renderStatus();
      intervalId = setInterval(tick, 1000);
    }

    function stopTimer() {
      if (intervalId) clearInterval(intervalId);
      intervalId = null;
      document.getElementById("status").textContent =
        "Parado. Clique em “Iniciar cronômetro”.";
    }
  </script>
